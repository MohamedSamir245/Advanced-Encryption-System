module x_time(
input[7:0] in_data,
output reg [7:0] out_data
);
reg [7:0] temp;
//initial
always@*
begin
					 temp=in_data<<1;
					out_data=(in_data[7]==1'b1) ?temp^8'h1b:temp;
end
endmodule

module multi(
input [7:0] in_data,
input [7:0] mat,
output reg [7:0] out_data
);
wire [7:0]temp;
wire [7:0]temp1;
x_time unit (in_data,temp1);
x_time unit1 (in_data,temp);
always@*
begin
	case(mat)
	8'h01 :	  out_data=in_data;
	8'h02 :    out_data=temp1;
	8'h03: begin
				out_data=temp^in_data;
			end
	//default  out_data=in_data;
	endcase
end
endmodule
